def existe_selon_pa(pa):
    p=pa//2
    for k1 in range(p//154 +1):
        p=pa//2-k1*154
        for k2 in range(p//116+1):
            p=pa//2-k1*154-k2*116
            for k3 in range(p//117+1):
                p=pa//2-k1*154-k2*116-k3*117
                for A in range(p//64+1):
                    p=pa//2-k1*154-k2*116-k3*117-A*64
                    for B in range(p//190+1):
                        p=pa//2-k1*154-k2*116-k3*117-B*190
                        for C in range(p//112+1):
                            if k1+k2+k3>=2*(A+B)-2:
                                if A+B>=2 and A+B-2>=C:
                                    if pa==2*(k1*154+k2*116+k3*117+A*64+B*190+C*112+(A+B-C-2)*156):
                                        return (k1,k2,k3,A,B,C)
    return False

def tourner_droite(l):
    n, m = len(l), len(l[0])
    t = [[0 for j in range(n)] for i in range(m)]
    for i in range(m):
        for j in range(n):
            t[i][j] = l[n-1-j][i]
    return t

def tourner_gauche(l):
    n, m = len(l), len(l[0])
    t = [[0 for j in range(n)] for i in range(m)]
    for i in range(n):
        for j in range(m):
            t[m-1-j][i] = l[i][j]
    return t


def etape(l,i,j):
    acc=-l[i][j]
    for x in range(i-1,i+2):
        for y in range(j-1,j+2):
            acc=acc+l[x][y]
    if l[i][j]==1 and (acc==2 or acc==3):
        return 1
    if l[i][j]==0 and acc==3:
        return 1
    return 0

def tousacote(l1):
    l=[(a[0],a[1]) for a in l1]
    for a in l1:
        i,j=a
        for x in range(i-1,i+2):
            for y in range(j-1,j+2):
                if (x,y) not in l:
                    l.append((x,y))
    return l


def etat_suivant(l,l1=[[]],ind=False):
    n,m=len(l),len(l[0])
    if l1==[[]]:
        l1=[(i,j) for i in range(n) for j in range(m)]
    t=[l[i[0]][i[1]] for i in l1 ]
    l2=[]
    for i in range(len(l1)) :
        t[i]=etape(l,l1[i][0],l1[i][1])
        if t[i]!=l[l1[i][0]][l1[i][1]]:
            l2.append((l1[i][0],l1[i][1]))
    for i in range(len(l1)):
        l[l1[i][0]][l1[i][1]] =t[i]
    if ind:
        return (l,l2)
    else:
        return l
